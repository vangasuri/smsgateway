create table smsgateway.SMS_ADMIN_USERS (
	ID int NOT NULL AUTO_INCREMENT,
	USER_NAME varchar(50) NOT NULL,
	PWD varchar(50) NOT NULL,
	USER_STATUS varchar(5) NOT NULL,
	USER_TYPE varchar(5) NOT NULL,
	PRIMARY KEY (ID)
);


create table smsgateway.SMS_PAPADAMS_CUST (
ID int NOT NULL AUTO_INCREMENT,
CUST_NAME varchar(50) NOT NULL,
PHNO varchar(15) NOT NULL,
BILL_AMNT double NOT NULL,
DISCOUNT double NOT NULL,
SMS_TRIGGERED varchar(10) NOT NULL,
PROMOTION_EXPIRY_DATE Date NOT NULL,
CREATED_BY varchar(50) NOT NULL,
PRIMARY KEY (ID)
);
